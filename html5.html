example
  KEY_ONLINE = lambda {|uuid| "user:online:#{uuid}"}
  KEY_ONLINES = lambda {|i| "onlines#{uuid}"}

  # for user online indication
  # expire in 5 minutes
  def self.update_online(user)
    Datastore.redis.set_with_expire(KEY_ONLINE.call(user.uuid), user.uuid, 300)

    # sepreate to keys: onlines0...onlines5
    i = Time.now.to_i / 600
    Datastore.redis.sadd(KEY_ONLINES.call(i), user.uuid)
  end

  def self.online?(user)
    !!Datastore.redis.get(KEY_ONLINE.call(user.uuid))
  end

  def self.onlines
    # sepreate to keys: onlines0...onlines5
    i = Time.now.to_i / 600
    Datastore.redis.smembers(KEY_ONLINES.call(i))
  end
